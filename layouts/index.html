  builds: ["software", "apps", "websites"],
  stack: ["TypeScript", "Go", "React"],
  cloud: "edge",
})

export async function ship(project) {
  const build = await rheinwork.compile(project)
  await deploy(build, { fast: true, scale: "auto" })
  return build.url
}
